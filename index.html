  <script>

    const compareToggle = document.getElementById('compareToggle');

    const compareConfig = document.getElementById('compareConfig');

    const compareOffTip = document.getElementById('compareOffTip');

    const fillToggle = document.getElementById('fillToggle');

    const compareCount = document.getElementById('compareCount');

    const compareStart = document.getElementById('compareStart');

    const mainAdType = document.getElementById('mainAdType');

    const summary = document.getElementById('summary');

    const tooltipBtn = document.getElementById('tooltipBtn');


    function sanitizeNumber(input, minValue) {

      const value = parseInt(input.value, 10);

      if (Number.isNaN(value) || value < minValue) {

        input.value = minValue;

      } else {

        input.value = value;

      }

    }


    function renderCompareArea() {

      const enabled = compareToggle.checked;

      compareConfig.classList.toggle('hidden', !enabled);

      compareOffTip.classList.toggle('hidden', enabled);

      renderSummary();

    }


    function renderSummary() {

      const adType = mainAdType.value;

      const fill = fillToggle.checked ? '开启' : '关闭';

      const compare = compareToggle.checked ? '开启' : '关闭';

      const count = compareCount.value;

      const start = compareStart.value;


      let html = '';

      html += `<span class="pill">主广告类型：${adType}</span>`;

      html += `<span class="pill">补位：${fill}</span>`;

      html += `<span class="pill">比价：${compare}</span>`;


      if (compareToggle.checked) {

        html += `<div style="margin-top:8px;">比价参数：单次session内最多比价 <b>${count}</b> 次；从第 <b>${start}</b> 次广告展示开始比价。</div>`;

      } else {

        html += '<div style="margin-top:8px;">当前关闭比价，仅按补位逻辑处理。</div>';

      }


      summary.innerHTML = html;

    }


    compareToggle.addEventListener('change', renderCompareArea);

    fillToggle.addEventListener('change', renderSummary);

    mainAdType.addEventListener('change', renderSummary);


    compareCount.addEventListener('input', () => {

      sanitizeNumber(compareCount, 1);

      renderSummary();

    });


    compareStart.addEventListener('input', () => {

      sanitizeNumber(compareStart, 0);

      renderSummary();

    });


    tooltipBtn.addEventListener('click', () => {

      tooltipBtn.classList.toggle('active');

    });


    document.addEventListener('click', (e) => {

      if (!tooltipBtn.contains(e.target)) {

        tooltipBtn.classList.remove('active');

      }

    });


    renderCompareArea();

  </script>
